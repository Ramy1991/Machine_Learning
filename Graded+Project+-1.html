def area_of_a_circle(user_in):
    if user_in.isdigit():
        R = int(user_in)
        PI = 3.1416
        return PI * R * R
    else:
        return 'Invalid input, please enter a number'

num_2 = 4

def check_prime(num):
    check = 0
    if num > 1:
        for i in range(2, num):
            if (num % i) == 0:
                check = 1
                break
    if check > 0:
        print(num, "is not a prime number")
    else:
        print(num, "is a prime number")
        
check_prime(num_1)     
check_prime(num_2)

a. 5.1 Write a function which will encrypt a given text. The function would require 2 arguments – the input text to be encrypted and a key. (5 points)
b. 5.2 Similarly create a decrypter which can decode the encrypted text when provided with the input text and key. (5 points)
c. For eg: Given the input text ‘hello’ and the key 3, the resulting encrypted text would be ‘khoor’. Here you can see that every character in the string hello is shifted by 3 characters. ‘h’ has shifted to ‘k’, ‘e’ has shifted to ‘h’ and so on. If a key of 5 were used, the resulting string would be ‘mjqqt’.
d. Example: encrypt(‘hello world!’, 3):- ‘khoor zruog!’
e. Example: decrypt(‘khoor zruog!’, 3):- ‘hello world!’

# word = input('Enter the plain text: ')
word = 'hello world!'

def cipher_encrypt(word, key):
    results = ''
    if type(key) != int:
        return 'The key must be an integer'
    else:
        for i in word:
            # Ignore number, special characters and Spaces
            if i.isalpha() == False:
                results += i
            else:
                # ord function to return the number of char
                results += chr(ord(i) + int(key))
    return results

cipher_encrypt(word, 3)

def cipher_decrypt(word, key):
    results = ''
    if type(key) != int:
        return 'The key must be an integer'
    else:
        for i in word:
            # Ignore number, special characters and Spaces
            if i.isalpha() == False:
                results += i
            else:
                # ord function to return the number of char

    return results
